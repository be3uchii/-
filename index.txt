async function hashPassword(password, username) {
  const salt = 'xchat_salt_' + username.toLowerCase() + '_2024';
  const data = new TextEncoder().encode(salt + password + salt);
  const buf = await crypto.subtle.digest('SHA-256', data);
  return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2,'0')).join('');
}

    regError.textContent = 'Слишком много попыток. Подождите ' + Math.ceil((registerBlockedUntil - now)/1000) + ' сек.';

    registerAttempts++;
    if (registerAttempts >= 5) { registerBlockedUntil = Date.now() + 60000; registerAttempts = 0; }

  const { data } = await sb.from('messages').select('*').order('created_at', { ascending: true }).limit(MSG_CACHE_LIMIT);
  if (data && data.length > 0) {
    messages = data; saveMessagesCache(messages); renderMessages(true); scrollToBottom();
  } else if (!cached || cached.length === 0) {
    messages = []; renderMessages(true);
  }
}

async function loadDmMessages() {
  if (!currentConversationId) return;
  const cached = loadDmCache(currentConversationId);
  if (cached && cached.length > 0) {
    dmMessages = cached; renderDmMessages(true); scrollToBottom();
  }
  const { data } = await sb.from('direct_messages').select('*')
    .eq('conversation_id', currentConversationId)
    .order('created_at', { ascending: true }).limit(MSG_CACHE_LIMIT);
  if (data && data.length > 0) {
    dmMessages = data; saveDmCache(currentConversationId, dmMessages); renderDmMessages(true); scrollToBottom();
  } else if (!cached || cached.length === 0) {
    dmMessages = []; renderDmMessages(true);
  }
}

function appendMsgEl(m, idx) {
  const isSelf = m.username === currentUser.username;

  const isCont = prevMsg && prevMsg.username === m.username;

  messagesArea.appendChild(el);
  return el;
}

function appendDmMsgEl(m, idx) {
  const isSelf = m.sender === currentUser.username;
  const prevMsg = idx > 0 ? dmMessages[idx - 1] : null;
  const isCont = prevMsg && prevMsg.sender === m.sender;
  const avatar = isSelf ? currentUser.avatarUrl : (currentDmUser ? currentDmUser.avatar_url : '');
  const username = isSelf ? currentUser.username : (currentDmUser ? currentDmUser.username : '');
  const msgForRender = { ...m, username, avatar_url: avatar };
  const el = buildMsgEl(msgForRender, isCont, isSelf);
  messagesArea.appendChild(el);

}

function subscribeGlobalRealtime() {
  realtimeChannel = sb.channel('chat-room')
    .on('postgres_changes', { event: 'INSERT', schema: 'public', table: 'messages' }, (payload) => {
      const msg = payload.new;
      if (!msg || !msg.username || !validUsername.test(msg.username)) return;
      if (!msg.image_url && (!msg.content || msg.content.length > 1000)) return;
      if (msg.image_url && msg.image_url !== '__pending__' && !msg.image_url.startsWith('https://')) return;
      const isMine = msg.username === currentUser.username;

      const tempIdx = messages.findIndex(m => m._temp && m.username === msg.username && (m.image_url ? !!msg.image_url : !msg.image_url));
      if (tempIdx !== -1) {
        messages[tempIdx] = msg; saveMessagesCache(messages);
        if (isMine) {
          const tempEl = messagesArea.querySelector('[data-id^="t_"]');
          if (tempEl) {
            tempEl.dataset.id = String(msg.id);
            if (msg.image_url) {
              const msgImg = tempEl.querySelector('.msg-img');
              if (msgImg && msg.image_url !== '__pending__') { imageUrlCache[msg.image_url] = msg.image_url; msgImg.src = msg.image_url; }
            }
          }
        }
        return;
      }
      if (messages.find(m => m.id === msg.id)) return;
      const atBottom = messagesArea.scrollTop + messagesArea.clientHeight >= messagesArea.scrollHeight - 80;
      messages.push(msg);
      if (messages.length > MSG_CACHE_LIMIT) messages = messages.slice(-MSG_CACHE_LIMIT);
      saveMessagesCache(messages);
      const es = messagesArea.querySelector('.empty-state');
      if (es) es.style.display = 'none';
      appendMsgEl(msg, messages.length - 1);
      if (atBottom) scrollToBottom();
    })
    .subscribe();
}

function makeCopyBtn(m) {
  const btn = document.createElement('button');
  btn.className = 'copy-btn';
  btn.title = 'Копировать';
  btn.innerHTML = COPY_ICON;
  btn.addEventListener('click', (e) => {
    e.stopPropagation();
    const text = m.username + ': ' + m.content;
    navigator.clipboard.writeText(text).then(() => {
      btn.innerHTML = CHECK_ICON;
      setTimeout(() => { btn.innerHTML = COPY_ICON; }, 1500);
    }).catch(() => {
      const ta = document.createElement('textarea');
      ta.value = text; ta.style.cssText = 'position:fixed;opacity:0;';
      document.body.appendChild(ta); ta.select(); document.execCommand('copy'); document.body.removeChild(ta);
      btn.innerHTML = CHECK_ICON;
      setTimeout(() => { btn.innerHTML = COPY_ICON; }, 1500);
    });
  });
  return btn;
}

function buildMsgEl(m, isCont, isSelf) {
  const av = avatarCache[m.avatar_url] || m.avatar_url || '';
  if (m.avatar_url && !avatarCache[m.avatar_url]) avatarCache[m.avatar_url] = m.avatar_url;

  const wrap = document.createElement('div');
  wrap.className = 'msg-group' + (isCont ? ' msg-continuation' : '') + (isSelf ? ' self' : '');
  wrap.dataset.id = m._tempId ? ('t_' + m._tempId) : String(m.id);

  const img = document.createElement('img');
  img.className = 'msg-avatar';
  img.src = av; img.alt = ''; img.loading = 'lazy';
  if (!isSelf && m.username && currentChatMode === 'global') {
    img.style.pointerEvents = 'auto';
    img.style.cursor = 'pointer';
    img.addEventListener('click', () => showProfilePopup(m.username, m.avatar_url));
  }
  wrap.appendChild(img);

  const body = document.createElement('div');
  body.className = 'msg-body';

  if (!isCont && !isSelf) {
    const uname = document.createElement('span');
    uname.className = 'msg-username';
    uname.textContent = m.username;
    body.appendChild(uname);
  }

  const bwrap = document.createElement('div');
  bwrap.className = 'msg-bubble-wrap';
  const timeStr = formatTime(m.created_at || new Date().toISOString());

  if (m.image_url === '__pending__' && m._tempImageDataUrl) {
    const bubble = document.createElement('div');
    bubble.className = 'msg-bubble has-image';
    const imgWrap = document.createElement('div');
    imgWrap.className = 'msg-img-wrap';
    const msgImg = document.createElement('img');
    msgImg.className = 'msg-img'; msgImg.src = m._tempImageDataUrl; msgImg.alt = '';
    imgWrap.appendChild(msgImg);
    const timeOverlay = document.createElement('span');
    timeOverlay.className = 'msg-img-time';
    timeOverlay.textContent = formatTime(m.created_at || new Date().toISOString());
    imgWrap.appendChild(timeOverlay);
    bubble.appendChild(imgWrap);
    bwrap.appendChild(bubble);
  } else if (m.image_url && m.image_url !== '__pending__') {
    const realSrc = imageUrlCache[m.image_url] || m.image_url;
    imageUrlCache[m.image_url] = realSrc;
    const bubble = document.createElement('div');
    bubble.className = 'msg-bubble has-image';
    const imgWrap = document.createElement('div');
    imgWrap.className = 'msg-img-wrap';
    const msgImg = document.createElement('img');
    msgImg.className = 'msg-img'; msgImg.src = m._tempImageDataUrl || realSrc; msgImg.alt = ''; msgImg.loading = 'lazy';
    imgWrap.addEventListener('click', () => { openLightbox(msgImg.src); });
    imgWrap.appendChild(msgImg);
    const timeOverlay = document.createElement('span');
    timeOverlay.className = 'msg-img-time'; timeOverlay.textContent = timeStr;
    imgWrap.appendChild(timeOverlay);
    bubble.appendChild(imgWrap);
    bwrap.appendChild(bubble);
  } else {
    const bubble = document.createElement('div');
    bubble.className = 'msg-bubble';
    const textSpan = document.createElement('span');
    textSpan.textContent = (m.content || '').trim();
    const timeSpan = document.createElement('span');
    timeSpan.className = 'msg-time'; timeSpan.textContent = timeStr;
    const isOneLiner = (m.content || '').split('\n').length === 1 && (m.content || '').length <= 28;
    if (isOneLiner) {
      bubble.style.cssText = 'display:inline-flex;align-items:flex-end;gap:5px;padding-right:8px;';
      textSpan.style.cssText = 'line-height:1.45;word-break:break-word;white-space:pre-wrap;min-width:0;';
      timeSpan.style.cssText = 'flex-shrink:0;padding-bottom:1px;';
    } else {
      bubble.style.cssText = 'display:block;padding-bottom:16px;';
      textSpan.style.cssText = 'line-height:1.45;word-break:break-word;white-space:pre-wrap;display:block;';
      timeSpan.style.cssText = 'position:absolute;bottom:4px;right:8px;';
    }
    bubble.appendChild(textSpan);
    bubble.appendChild(timeSpan);
    bwrap.appendChild(bubble);
    if (!m._temp && !isSelf) bwrap.appendChild(makeCopyBtn(m));
  }

  body.appendChild(bwrap);
  wrap.appendChild(body);
  return wrap;
}

async function sendMessage() {
  const content = msgInput.value.trim();
  if ((!content && !chatImageFile) || sending) return;

  const now = Date.now();
  if (now - lastMessageTime < MSG_COOLDOWN_MS) return;
  if (now > messageCountResetTime) { messageCount = 0; messageCountResetTime = now + MSG_RATE_WINDOW; }
  messageCount++;
  if (messageCount > MSG_RATE_LIMIT) return;
  lastMessageTime = now;
  if (content.length > 1000) return;

  sending = true;
  const imageFile = chatImageFile;
  const imageDataUrl = chatImageDataUrl;
  const tempId = ++_tempIdCounter;

  msgInput.value = ''; msgInput.style.height = '22px'; msgInput.style.overflow = 'hidden';
  msgInput.disabled = false; msgInput.placeholder = 'Сообщение...';
  btnSend.disabled = true; msgInput.blur();
  chatImageFile = null; chatImageDataUrl = null;
  imagePreviewThumb.src = '';
  imagePreviewBar.classList.remove('visible');
  btnAttach.classList.remove('has-image');

  if (currentChatMode === 'global') {
    await sendGlobalMessage(content, imageFile, imageDataUrl, tempId);
  } else {
    await sendDmMessage(content, imageFile, imageDataUrl, tempId);
  }

  sending = false;
}

async function sendGlobalMessage(content, imageFile, imageDataUrl, tempId) {
  const tempMsg = {
    _temp: true, _tempId: tempId,
    username: currentUser.username, avatar_url: currentUser.avatarUrl,
    content: imageFile ? ' ' : content,
    image_url: imageFile ? '__pending__' : null,
    _tempImageDataUrl: imageDataUrl || null,
    created_at: new Date().toISOString()
  };
  messages.push(tempMsg);
  const es = messagesArea.querySelector('.empty-state');
  if (es) es.style.display = 'none';
  appendMsgEl(tempMsg, messages.length - 1);
  scrollToBottom();

  let image_url = null;
  if (imageFile) {
    const allowedExts = ['jpg','jpeg','png','webp'];
    const ext = imageFile.name.split('.').pop().toLowerCase().replace(/[^a-z]/g,'');
    if (!allowedExts.includes(ext) || !ALLOWED_IMAGE_TYPES.includes(imageFile.type)) { sending = false; return; }
    const path = currentUser.username + '_' + Date.now() + '.' + ext;
    const { error: uploadErr } = await sb.storage.from('chat-images').upload(path, imageFile, { cacheControl: '31536000', upsert: false });
    if (!uploadErr) {
      const { data: urlData } = sb.storage.from('chat-images').getPublicUrl(path);
      image_url = urlData.publicUrl;
    }
  }

  const { data: inserted, error: insertErr } = await sb.from('messages').insert({
    username: currentUser.username, avatar_url: currentUser.avatarUrl || null,
    content: imageFile ? ' ' : content, image_url
  }).select().single();

  if (insertErr) { console.error('Insert error:', insertErr); }
  if (inserted) {
    const tempIdx = messages.findIndex(m => m._tempId === tempId);
    if (tempIdx !== -1) {
      messages[tempIdx] = inserted; saveMessagesCache(messages);
      const tempEl = messagesArea.querySelector('[data-id="t_' + tempId + '"]');
      if (tempEl) {
        tempEl.dataset.id = String(inserted.id);
        if (inserted.image_url) {
          const msgImg = tempEl.querySelector('.msg-img');
          if (msgImg) { imageUrlCache[inserted.image_url] = inserted.image_url; msgImg.src = inserted.image_url; }
        }
      }
    }
  }
}

async function sendDmMessage(content, imageFile, imageDataUrl, tempId) {
  let convId = currentConversationId;

  if (!convId) {
    const u1 = currentUser.username < currentDmUser.username ? currentUser.username : currentDmUser.username;
    const u2 = currentUser.username < currentDmUser.username ? currentDmUser.username : currentUser.username;
    const { data: existing } = await sb.from('conversations').select('id').eq('user1', u1).eq('user2', u2).maybeSingle();
    if (existing) {
      convId = existing.id;
    } else {
      const { data: newConv } = await sb.from('conversations').insert({ user1: u1, user2: u2 }).select().single();
      if (newConv) convId = newConv.id;
    }
    currentConversationId = convId;
    subscribeDmRealtime();
  }

  if (!convId) { sending = false; return; }

  const tempMsg = {
    _temp: true, _tempId: tempId,
    sender: currentUser.username,
    content: imageFile ? ' ' : content,
    image_url: imageFile ? '__pending__' : null,
    _tempImageDataUrl: imageDataUrl || null,
    conversation_id: convId,
    created_at: new Date().toISOString()
  };
  dmMessages.push(tempMsg);
  const es = messagesArea.querySelector('.empty-state');
  if (es) es.style.display = 'none';
  appendDmMsgEl(tempMsg, dmMessages.length - 1);
  scrollToBottom();

  let image_url = null;
  if (imageFile) {
    const allowedExts = ['jpg','jpeg','png','webp'];
    const ext = imageFile.name.split('.').pop().toLowerCase().replace(/[^a-z]/g,'');
    if (!allowedExts.includes(ext) || !ALLOWED_IMAGE_TYPES.includes(imageFile.type)) { sending = false; return; }
    const path = 'dm_' + currentUser.username + '_' + Date.now() + '.' + ext;
    const { error: uploadErr } = await sb.storage.from('chat-images').upload(path, imageFile, { cacheControl: '31536000', upsert: false });
    if (!uploadErr) {
      const { data: urlData } = sb.storage.from('chat-images').getPublicUrl(path);
      image_url = urlData.publicUrl;
    }
  }

  const { data: inserted } = await sb.from('direct_messages').insert({
    conversation_id: convId, sender: currentUser.username,
    content: imageFile ? ' ' : content, image_url
  }).select().single();

  if (inserted) {
    const tempIdx = dmMessages.findIndex(m => m._tempId === tempId);
    if (tempIdx !== -1) {
      dmMessages[tempIdx] = inserted; saveDmCache(convId, dmMessages);
      const tempEl = messagesArea.querySelector('[data-id="t_' + tempId + '"]');
      if (tempEl) {
        tempEl.dataset.id = String(inserted.id);
        if (inserted.image_url) {
          const msgImg = tempEl.querySelector('.msg-img');
          if (msgImg) { imageUrlCache[inserted.image_url] = inserted.image_url; msgImg.src = inserted.image_url; }
        }
      }
    }
    await sb.from('conversations').update({
      last_message: imageFile ? null : content,
      last_message_image: !!imageFile,
      last_message_at: inserted.created_at
    }).eq('id', convId);
  }
}
